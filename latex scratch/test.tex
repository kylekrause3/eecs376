\begin{algorithm}[H]
    \begin{algorithmic}[1]
        \Require{a natural number $n$}
        \Ensure{the $n$-th Fibonacci number}
        \Function{\text{Fib}}{$n$}
            \If{$n = 0$}
                \State \Return $0$
            \EndIf
            \If{$n = 1$}
                \State \Return $1$
            \EndIf
            \State \Return $\text{Fib}(n-1) + \text{Fib}(n-2)$
        \EndFunction
    \end{algorithmic}
\end{algorithm}
